\section{Metodologia}

\subsection{Informações do modelo proposto no periódico}

\begin{frame}
\frametitle{Automato Celular}
\begin{Large}
Definição: Um Automato Celular (AC) é uma sistema dinâmico, discreto em espaço e tempo.
\end{Large}\\
\end{frame}

\begin{frame}
\frametitle{Automato Celular}
\begin{large}
Representação da Rede Bidimensional do Automato Celular:
\begin{equation}
N^{t} = \{\overrightarrow{n_{1}},\overrightarrow{n_{2}},...,\overrightarrow{n_{m}}\}
\end{equation}

\begin{equation}
\overrightarrow{n_{d}} \in \mathbb{Z}^{2}, n_{d} \neq n_{f}
\end{equation}
e $m$ é o número máximo de células, sabemos que
\begin{equation}
\overrightarrow{n_{1}} = (n_{i},n_{j})
\end{equation}
Regra de vizinhança adotada no periódico foi a de \textbf{Moore}, ou seja, considera a vizinhança de 8 vizinhos, e a topologia da fronteira é a de uma esfera.
\end{large}
\end{frame}

\begin{frame}
\frametitle{Automato Celular}
\begin{large}
\begin{equation}
\overrightarrow{n_{1}} = (n_{i},n_{j}) \therefore {X}^{t} = \{{X_{1}}^{t},{X_{2}}^{t}\}
\end{equation}
Posição:
\begin{equation}
{X}^{t} = \{{X_{1}}^{t},{X_{2}}^{t}\}
\end{equation}
Velocidade:
\begin{equation}
{V}^{t} = \{{V_{1}}^{t},{V_{2}}^{t}\}
\end{equation}
\end{large}
\end{frame}

\begin{frame}
\frametitle{Algorítimo Otimização por Enxame de Partículas (PSO)}
\begin{large}
Velocidade:
\begin{equation}
V_{l}^{t+1} = \textcolor[rgb]{0.00,0.00,0.69}{{\omega}{V_{l}}^{t}}+\textcolor[rgb]{0.00,0.69,0.00}{{k_{1}}{q1}({{P_{l}}^t}-{{X_{l}}^t})}+\textcolor[rgb]{0.69,0.00,0.00}{{k_{2}}{q2}({{P_{g}}^t}-{{X_{l}}^t})}
\end{equation}
Posição:
\begin{equation}
{X_{l}}^{t+1} = {X_{l}}^{t}+{V_{l}}^{t+1}
\end{equation}
\end{large}
\begin{itemize}
\item $\omega$: Inércia da partícula
\item $q_{1}$: Número aleatório entre [0,1]
\item $q_{2}$: Número aleatório entre [0,1]
\item $k_{1}$: Fator cognitiva
\item $k_{2}$: Fator social
\item $P_{l}$: Melhor posição de adaptação
\item $P_{g}$: Melhor posição global de adaptação
\end{itemize}
\end{frame}
